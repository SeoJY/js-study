Study Summary - #04

while문

  • 변수선언;
    while (조건문) { 증감연산자 }
var a = 0;
while (a < 10) {
    console.log(a);
    a++;
}
var a = Math.random() + 10;
while (a === 5) {
    console.log(a);
    a++;
}

for문

  • for(변수선언; 조건문; 증감연산자) { }
for(var i = 0; i < 10; i++){
    console.log(i);
}
  • for문 조건문에 $('li').length같은 방식으로 사용하면 최적화에 좋지 않음.
  • 변수에 넣고 조건문에 넣는 것이 좋음.
for(var j = 0, max = $('li').length; j < max; j++){
    console.log(j);
}

array와 for문의 결합

var arrayObj = [
    ['data1', 'data2', 'data3', 'data4', 'data5'],
    ['attribute1', 'attribute2', 'attribute3', 'attribute4', 'attribute5'],
    ['data1', 'data2', 'data3', 'data4', 'data5'],
    ['attribute01', 'attribute02', 'attribute03', 'attribute04', 'attribute05'],
],
    tagPush = ''; // 문자열
for (var i = 0, max = arrayObj.length; i < max; i++) {
    tagPush += '<tr>';
    for(var td = 0; td < 5; td++) {
        tagPush += '<td>' + arrayObj[i][td] + '</td>'
    }
    tagPush += '</tr>';
}
$('#data_tbody').get(0).innerHTML = tagPush;
